class Solution {
public:
    int maxScoreSightseeingPair(vector<int>& values) {
        int maxscore=0;
        int m=values[0];
        for(int i=1;i<values.size();i++){
           if(values[i]-i+m > maxscore){
            maxscore= values[i]-i+m;
           }
           if(values[i]+i>m){
            m=values[i]+i;
           }
        }
        return maxscore;
    }
};
